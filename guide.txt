# 1. Make sure metrics-server is working
kubectl top nodes
kubectl top pods

# 2. Apply your app + HPA files
kubectl apply -f kubernetes/

# 3. Make HPA sensitive for demo
kubectl patch hpa backend-hpa \
-p '{"spec":{"metrics":[{"type":"Resource","resource":{"name":"cpu","target":{"type":"Utilization","averageUtilization":10}}}]}}'

# 4. Check initial HPA state
kubectl get hpa
kubectl get pods
kubectl top pods

# 5. Create load on backend
kubectl run backend-loader-1 --image=busybox --restart=Never -- /bin/sh -c "while true; do wget -q -O- http://backend-service:5000/api/products; done"

kubectl run backend-loader-2 --image=busybox --restart=Never -- /bin/sh -c "while true; do wget -q -O- http://backend-service:5000/api/products; done"

kubectl run backend-loader-3 --image=busybox --restart=Never -- /bin/sh -c "while true; do wget -q -O- http://backend-service:5000/api/products; done"

# 6. Watch autoscaling
kubectl get hpa
kubectl get pods
kubectl top pods

# Optional live watch
kubectl get hpa -w


# Stop watching first with CTRL + C if using -w

kubectl delete pod backend-loader-1 backend-loader-2 backend-loader-3 --ignore-not-found

kubectl patch hpa backend-hpa \

kubectl get hpa
